def to_s
  if @bridge_port == true
    "Bridge: [#{@port_name}, #{@remote_switch_name}, #{@remote_port_name}, #{@remote_port_number}]"
  else
    "Std Port: [#{@port_name}, [ #{@port_arp.join(',')} ] ]"
  end
end
